\section*{3. Dimensional Analysis and Estimates}

\subsection*{3.1 Gravitational Sector}

\textbf{Mass-to-index coupling:}



\[
\mu_G = \frac{2G}{c^2} = 1.486 \times 10^{-27}\,\text{m}^2/\text{kg}.
\]



Validated by matching GR weak-field predictions.

\textbf{Mass-current-to-flow coupling:}



\[
A_M = \frac{G}{c^2} = 7.426 \times 10^{-28}\,\text{m}^3/(\text{kg}\cdot\text{s}).
\]



Constrained by GR gravitomagnetic analogy (Lense--Thirring).

\subsection*{3.2 Electromagnetic Sector}

\textbf{Charge-to-index coupling:}

Dimensional estimate from classical electron radius:



\[
\mu_E \sim \frac{e^2}{4\pi\varepsilon_0 m_e c^2}
\sim 2.82 \times 10^{-15}\,\text{m (electron radius)}.
\]



But this is for electrons. For general charge-to-index:



\[
\mu_E \sim \frac{e^2}{\varepsilon_0 c^2}
\sim 2.54 \times 10^{-37}\,\text{m}^2/\text{C}.
\]



\textbf{Status:} Unconstrained. Requires charged-mass lensing experiments.

\textbf{Charge-current-to-flow coupling:}

Dimensional estimate:



\[
A_q \sim \frac{e^2}{\varepsilon_0 c^2}
\sim 2.54 \times 10^{-37}\,\text{m}^3/(\text{C}\cdot\text{s}).
\]



\textbf{Observational upper bound} from lab magnet null results (1 MA solenoid,
10 cm distance, $< 10^{-10}$ rad deflection threshold):



\[
A_q \lesssim 3 \times 10^{-10}\,\text{m}^3/(\text{C}\cdot\text{s}).
\]



This is $27$ orders of magnitude \emph{larger} than dimensional estimate but
still $17$ orders of magnitude \emph{smaller} than $A_M$.

\textbf{Practical assumption:} $A_q \approx 0$.

\subsection*{3.3 Plasma Sector}

\textbf{Electron-density-to-index coupling:}

From standard plasma optics $n \approx 1 - \omega_p^2/(2\omega^2)$:



\[
\alpha_{\text{plasma}}
\sim \frac{e^2}{2\varepsilon_0 m_e \omega^2}
\sim 10^{-15}\,\text{m}^3

\textbf{Status:} Standard plasma physics. PM extends it to modify gravitational

\section*{4. Observational Constraints: Detailed Derivations}

\subsection*{4.1 Mass Current Coupling ($A_M$)}

\textbf{Test:} GR frame-dragging (Gravity Probe B, LAGEOS satellites).

GR gravitomagnetic potential:



\[
h_i^{\text{GR}} \sim \frac{G M v_i}{c^2 r}.
\]



PM flow field:



\[
u_i^{\text{PM}} = A_M M \frac{v_i}{r}.
\]



Matching weak-field predictions:



\[
A_M = \frac{G}{c^2}.
\]



\textbf{Constraint:} $A_M = (7.426 \pm 0.001) \times 10^{-28}$ m$^3$/(kg$\cdot$s)
(precision limited by GR test accuracy).

\subsection*{4.2 Charge Current Coupling ($A_q$)}

\textbf{Test:} Null results from optical systems near laboratory electromagnets.

\textbf{Setup:} Superconducting solenoid, $I = 10^6$ A, $L = 1$ m, observation
distance $r = 0.1$ m, path length $\ell_{\text{path}} = 1$ m.

Flow field magnitude:



\[
|\mathbf u_q| \sim A_q\,I\,\frac{L}{r}.
\]



Angular deflection:



\[
\alpha \sim \frac{|\mathbf u_q|}{c}\,\frac{\ell_{\text{path}}}{r}
= A_q\,I\,\frac{L\,\ell_{\text{path}}}{c\,r^2}.
\]



Demanding $\alpha < 10^{-10}$ rad:



\[
A_q < \frac{10^{-10}\,c\,r^2}{I\,L\,\ell_{\text{path}}}
= \frac{10^{-10} \times 3\times10^8 \times 0.01}{10^6 \times 1 \times 1}
\sim 3 \times 10^{-10}\,\text{m}^3/(\text{C}\cdot\text{s}).
\]



\textbf{Constraint:} $A_q < 3 \times 10^{-10}$ m$^3$/(C$\cdot$s).

Ratio to gravitational coupling:



\[
\frac{A_q}{A_M} < 4 \times 10^{17}.
\]



\subsection*{4.3 Magnetic Potential Coupling ($\kappa_B$)}

\textbf{Test:} Same null results as $A_q$.

If $\mathbf u_B = \kappa_B\,\mathbf A$ with:



\[
|\mathbf A| \sim \frac{\mu_0 I L}{4\pi r},
\]



then with $\kappa_B \sim c$:



\[
|\mathbf u_B| \sim \frac{c\,\mu_0 I L}{4\pi r}
\sim 10^8\,\text{m/s}
\sim 0.3\,c
\]



for the solenoid parameters above. This would give $\alpha \sim 10$ rad.

\textbf{Observational verdict:} Not seen.

\textbf{Constraint:}



\[
\kappa_B < 10^{-10}\,c,
\]



or effectively:



\[
\kappa_B \approx 0.
\]



\textbf{Structural resolution:} Drop $\kappa_B\,\mathbf A$ term entirely. Couple
only to source currents $\mathbf J_q$ via $A_q$.
