\section{Core Assumptions}
%====================================================================

\begin{enumerate}\setlength{\itemsep}{2pt}
  \item \textbf{Flat spacetime.}
        Space is a compressible physical medium on a fixed Euclidean background.
        No GR curvature, no Einstein equations, no horizons, no TOV.

  \item \textbf{Scalar compression field.}
        Gravity arises from the scalar field $\phi$ with refractive index
        $n=e^{\phi}$.  Sources: $\nabla^2\phi = -\tfrac{8\pi G}{c^2}\rho$.

  \item \textbf{Density law.}
        \[ \rho = \rho_{\mathrm{nuc}}\,e^{\phi} \]

  \item \textbf{Force law for massive particles.}
        \[ \mathbf{a} = \tfrac{c^2}{2}\,\nabla\phi \]
        (Half the photon coupling; recovers $-GM/r^2\,\hat{\mathbf{r}}$ exactly.)

  \item \textbf{Light propagation.}
        Follows Fermat's principle in the optical metric:
        \[ ds_{\mathrm{opt}}^2 = -\frac{c^2}{n^2}\,dt^2
           + (d\mathbf{r} - \mathbf{u}\,dt)^2 \]

  \item \textbf{Self-consistent EOS.}
        \[ P = \frac{c^2}{2}\!\left(\rho - \rho_{\mathrm{nuc}}\right) \]

  \item \textbf{Stability cap.}
        The medium cannot be compressed beyond $\phi = 1$ ($n = e$,
        $\rho = e\,\rho_{\mathrm{nuc}}$).  This is the only strong-field limit
        in PM; it is not a compactness bound from GR.
\end{enumerate}

Data: 14 published RSD $f\sigma_8$ measurements, $z=0.02$--$1.40$
(6dFGS, SDSS MGS, WiggleZ, BOSS LOWZ/CMASS, FastSound, etc.).

\textbf{Key conclusions:}
\begin{enumerate}
  \item PM-Drag with $\Omega_{\rm eff}=0.315$ achieves marginally \emph{lower}
        $\chi^2/N$ than $\Lambda$CDM on current data, but only because it
        effectively uses $\Omega_{\rm eff}$ as a free parameter separately from
        the $D_L$ calibration.  There is \emph{no} parameter-free PM prediction.
  \item PM-Static is excluded at $\chi^2/N\approx1.6$ (wrong slope).
  \item PM with $\Omega_{\rm eff}=1$ (no dark energy) is excluded by ODE failure:
        it cannot reproduce a matter-dominated era.
  \item A joint fit $[\beta, \Omega_{\rm eff}, \sigma_{8,0}]$ constrained
        simultaneously by $D_L$ supernovae + $f\sigma_8$ RSD surveys would
        tightly constrain or rule out the refractive model.
\end{enumerate}
